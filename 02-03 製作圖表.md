```
import numpy as np
import matplotlib.pyplot as plt

#--------------------------------------------------------------------------------------------
# 讀取鳶尾花資料
# 花萼長, 花萼寬, 花瓣長, 花瓣寬, 花種編號 (山鳶尾花:1, 變色鳶尾花:2, 維吉尼亞鳶尾花:3)
#--------------------------------------------------------------------------------------------
data=np.genfromtxt('iris.csv',  names=['sepal_length', 'sepal_width', 'petal_length', 'petal_width', 'classes'], delimiter=',')


#---------------------------
# 繪圖
#---------------------------
fig = plt.figure()

# 左上子圖
feature_1='sepal_length'
feature_2='sepal_width'
sub = fig.add_subplot(221)
sub.plot(data[feature_1][data['classes']==1], data[feature_2][data['classes']==1], 'bd', alpha=0.2)
sub.plot(data[feature_1][data['classes']==2], data[feature_2][data['classes']==2], 'ro', alpha=0.2)
sub.plot(data[feature_1][data['classes']==3], data[feature_2][data['classes']==3], 'gv', alpha=0.2)

# 右上子圖
feature_1='petal_length'
feature_2='petal_width'
sub = fig.add_subplot(222)
sub.plot(data[feature_1][data['classes']==1], data[feature_2][data['classes']==1], 'bd', alpha=0.2)
sub.plot(data[feature_1][data['classes']==2], data[feature_2][data['classes']==2], 'ro', alpha=0.2)
sub.plot(data[feature_1][data['classes']==3], data[feature_2][data['classes']==3], 'gv', alpha=0.2)

# 左下子圖
feature_1='sepal_length'
feature_2='petal_length'
sub = fig.add_subplot(223)
sub.plot(data[feature_1][data['classes']==1], data[feature_2][data['classes']==1], 'bd', alpha=0.2)
sub.plot(data[feature_1][data['classes']==2], data[feature_2][data['classes']==2], 'ro', alpha=0.2)
sub.plot(data[feature_1][data['classes']==3], data[feature_2][data['classes']==3], 'gv', alpha=0.2)

# 右下子圖
feature_1='sepal_width'
feature_2='petal_width'
sub = fig.add_subplot(224)
sub.plot(data[feature_1][data['classes']==1], data[feature_2][data['classes']==1], 'bd', alpha=0.2)
sub.plot(data[feature_1][data['classes']==2], data[feature_2][data['classes']==2], 'ro', alpha=0.2)
sub.plot(data[feature_1][data['classes']==3], data[feature_2][data['classes']==3], 'gv', alpha=0.2)

#---------------------------
# 儲存圖檔
#---------------------------
fig.savefig('graph.png')
```
